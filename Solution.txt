import java.util.Scanner;
public  class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int N = Integer.parseInt(scanner.nextLine());
        String word = "";
        for (int i = 1; i <=N; i++) {
            char character = scanner.nextLine().charAt(0);
            word+=Encrypt(character);
        }
        System.out.println(word);
    }

    static String Encrypt(char letter) {
       int asciiCode= (int)letter;
        String finalWord = "";
        int secondDigit = asciiCode%10;
        int firstDigit = GetForstDogot(asciiCode);
        char firstLetter = (char)(letter+secondDigit);
        char lastLetter = (char)(letter-firstDigit);
        finalWord =firstLetter+Integer.toString(firstDigit)+secondDigit+lastLetter;
        return finalWord;
    }

    static int GetForstDogot(int asciiCode) {
        while(asciiCode>=10){
            asciiCode/= 10;
        }
        return  asciiCode;
    }
}
